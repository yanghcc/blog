# 如何实现vue组件重载

# 一、利用v-if控制

<template>
  <router-view v-if="isRouterAlive"/>
</template>
<script>
export default {
 data () {
   return {
     isRouterAlive: true
   }
 },
 methods: {
   reload () {
     this.isRouterAlive = false
     this.$nextTick(() => (this.isRouterAlive = true))
   }
 }
}
</script>

// 在子组件中注入父组件中的变量在data（）之前注入
inject:['reload']
// 然后其它任何想刷新自己的路由页面，都可以这样：
this.reload()
